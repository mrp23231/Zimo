```typescript
const uploadBlobToStorage = async (path, blob, onProgress, timeoutMs = 300000) => {
  let attempt = 0;
  while (attempt <= maxRetries) {
    try {
      return await uploadWithRetry();
    } catch (error) {
      if (!shouldRetry(error)) break;
      await delay(2^attempt * 1000); // 1s, 2s, 4s
    }
  }
}
```
